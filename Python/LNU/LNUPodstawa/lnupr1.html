    4. Ćwiczenie - print()
    #wyswietl tekst "Monty Python's Flying Circus"

    5. Wyrażenia
    #wyswietl wartość wyrażenia 1 + 2 + 3

    6. Operatory arytmetyczne
    #wyświetl pole powierzchni prostokąta o wymiarach 17 na 23

    7. Ćwiczenie - bagietki
    print(2.34 * 14)

    8. Print() - kilka wyrażeń
    #wyświetl komunikat zgodny z trescią zadania
    print(f"Koszt jednej parówki to: {4.50 / 6} PLN")

    9. Ćwiczenie - parówki
    print(f"Każde dziecko otrzyma {133 // 16} parówek")

    10. Ćwiczenie - modulo
    print(f"Dla nauczyciela zostało {133 % 16} parówek")

    11. Zmienne
    price = 0.75
    print(f'Parówka kosztuje: {price} PLN')

    12. Ćwiczenie - uczniowie
    students = 16
    print ("Na pikniku było", students, "uczniów")

    13. Metody
    text = "ave caesar!"
    
    14. lower()
    text = "MORITURI TE SALUTANT"
    print(text.lower() )

    15. capitalize()
    quote = "two things are infinite: the universe and human stupidity; and I'm not sure about the universe."
    print(quote.capitalize() )

    16. title()
    some_title = "monty python's flying circus"
    print(some_title.title() )
    
    17. replace()
    some_text = "Mmmmm, xxx, yummy xxx, I love xxx!"
    result_text = some_text.replace("xxx", "sausages")
    print(result_text)

    18. swapcase()
    quote = "The hardest thing in the world to understand is the Income Tax."
    print(quote.swapcase() )
    
    19. strip()
    quote = "     just a flesh wound.\n"
    print(quote.strip() )
    
    20. Konkatenacja
    text1 = "monty"
    text2 = "python"
    print(text1 + text2)

    21. Ćwiczenie - imię i nazwisko
    first_name = "clint"
    last_name = "eastwood"
    result = first_name.capitalize() + ", " + last_name.capitalize()
    print(result)

    22. Mnożenie stringów
    text1 = "Ole! "
    print(text1 * 10)
    
    23. Parametr sep
    nr1 = 123
    nr2 = 456
    nr3 = 789
    print(nr1, nr2, nr3, sep=".")

    24. Parametr end
    print("Black",end=" * ")
    print("Knight")
    
    25. Combo
    actor1 = "clint eastwood"
    actor2 = "bee vang"
    actor3 = "scott eastwood"
    print("Gran Torino cast: ", end="")
    print(actor1.title(),actor2.title(),actor3.title(), sep=", ", end=".")
