    13. Ćwiczenie: udział znaku
    def get_percentage(value, text):
      if len(text) == 0:
          return 0.0
      
      count = text.count(value)
      percentage = (count / len(text)) * 100.0
      return percentage

    16. Funkcja ord()
    def maxOrd(text):
      max_value = 0
      for char in text:
          char_value = ord(char)
          if char_value > max_value:
              max_value = char_value
      return max_value

      decrypted_message = ""

          decrypted_char_code = char_code // key
