          return 1

      length = 0
      for line in czytaj:
          length += 1
      
      czytaj.seek(0)
      l = 0
      
      for line in czytaj:
          l += 1
          line = line.split()
          for data in line:
              try:
                  data = int(data)
              except ValueError:
                  return 2
  
              if data < 0 or data > 127:
                  return 3
  
              pisz.write(chr(data))
          if l < length:

      return 0
